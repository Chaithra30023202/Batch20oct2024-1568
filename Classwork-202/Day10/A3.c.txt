#include <stdlib.h>
 
int num(int n) {
    int count = 0;
    for (int i = 2; ; i++) {
        int num = i;
        int isPd = 1;

        while (num > 0) {
            int digit = num % 10;
            if (digit != 2 && digit != 3 && digit != 5 && digit != 7) {
                isPd= 0;
                break;
            }
            num /= 10;
        }
 
        if (isPd) {
            count++;
            if (count == n) {
                return i;
            }
        }
    }
}
 
int main() {
    int T, N;
    printf("Enter the number of test cases: ");
    scanf("%d", &T);
 
    while (T--) {
        printf("Enter the value of N: ");
        scanf("%d", &N);
        int result = num(N);
        printf("%d\n", result);
    }
 
    return 0;
}
